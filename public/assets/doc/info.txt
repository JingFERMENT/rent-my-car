SELECT `categories`.`name` AS 'category_name', COUNT(`Vehicles`.`id_vehicle`) AS 'nb_vehicles'
FROM `Categories`
LEFT JOIN `Vehicles` ON `categories`.`id_category` = `vehicles`.`id_category`
GROUP BY `categories`.`name`
HAVING (`nb_vehicles`>2);
